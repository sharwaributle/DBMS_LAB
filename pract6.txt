(
    a IN NUMBER,
    b IN NUMBER, c OUT NUMBER
)
AS
BEGIN
    c:= a + b;
    DBMS_OUTPUT.PUT_LINE('Result: ' || c);
END;
/

variable c1 number;
 execute addition3(5,10, :c1);

CREATE OR REPLACE PROCEDURE addition4
(
   a IN OUT NUMBER, 
   b IN OUT NUMBER
)
AS
BEGIN
   a := a + b;
   b := a - b;
   a := a - b;
END;
/

=> VARIABLE a NUMBER;
VARIABLE b NUMBER;

BEGIN
   :a := 5;
   :b := 10;
END;
/

EXEC addition4(:a, :b);

PRINT a;
PRINT b;

CREATE OR REPLACE PROCEDURE LARGEST
(
   a IN NUMBER, 
   b IN NUMBER,
   c IN NUMBER,
   LARGEST OUT NUMBER
)
IS
BEGIN
IF a >b AND a> c THEN
   LARGEST:= a;
ELSEIF b > a AND b > c THEN
   LARGEST:= b;
ELSE LARGEST:=c;
END IF;
END LARGEST;
/
=>DECLARE
   x NUMBER := 10;
   y NUMBER := 25;
   z NUMBER := 15;
   result NUMBER;
BEGIN
   LARGEST(x, y, z, result);
   DBMS_OUTPUT.PUT_LINE('Largest number is: ' || result);
END;
/
